    readonly latitude: number;
    readonly longitude: number;

    constructor(latitude: number, longitude: number) {
        if (latitude < -90 || latitude > 90) {

        }
        if (longitude < -180 || longitude > 180) {

        }
        this.latitude = latitude;
        this.longitude = longitude;
    }
}
